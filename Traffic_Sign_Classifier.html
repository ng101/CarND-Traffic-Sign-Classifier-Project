def resample(X, y, min_samples_per_class):
    labels = np.unique(y)
    
    X_resampled = []
    y_resampled = []
    
    for i in range(0, len(labels)):
        Xi = X[y==labels[i]]
        yi = y[y==labels[i]]
        if len(yi) < min_samples_per_class:
            sample = np.random.choice(len(Xi), size=min_samples_per_class, replace=True)
            Xi = Xi[sample]
            yi = yi[sample]
        X_resampled.append(Xi)
        y_resampled.append(yi)
    return np.concatenate(X_resampled), np.concatenate(y_resampled)

import tensorflow as tf
from tensorflow.contrib.layers import flatten

def Net(x):    
    # Hyperparameters
    mu = 0
    sigma = 0.1
    
    # TODO: Layer 1: Convolutional. Input = 32x32x1. Output = 32x32x8.
    W = tf.Variable(tf.truncated_normal((2,2,1,8), mean=mu, stddev=sigma))
    B = tf.Variable(tf.zeros(8))
    out = tf.nn.conv2d(x, W, [1, 1, 1, 1], 'SAME') + B

    # TODO: Activation.
    out = tf.nn.relu(out)
    
    # TODO: Layer 2: Convolutional. Input = 32x32x8. Output = 28x28x16.
    W = tf.Variable(tf.truncated_normal((5,5,8,16), mean=mu, stddev=sigma))
    B = tf.Variable(tf.zeros(16))
    out = tf.nn.conv2d(out, W, [1, 1, 1, 1], 'VALID') + B

    # TODO: Pooling. Input = 28x28x16. Output = 14x14x16.
    out = tf.nn.max_pool(out, [1, 2, 2, 1], [1, 2, 2, 1], 'VALID')

    # TODO: Layer 3: Convolutional. Output = 10x10x32.
    W = tf.Variable(tf.truncated_normal((5, 5, 16, 32), mean=mu, stddev=sigma))
    B = tf.Variable(tf.zeros(32))
    out = tf.nn.conv2d(out, W, [1, 1, 1, 1], 'VALID') + B
    
    # TODO: Activation.
    out = tf.nn.relu(out)

    # TODO: Pooling. Input = 10x10x32. Output = 5x5x32.
    out = tf.nn.max_pool(out, [1, 2, 2, 1], [1, 2, 2, 1], 'VALID')

    # TODO: Flatten. Input = 5x5x32. Output = 800.
    out = flatten(out)
    
    # TODO: Layer 4: Fully Connected. Input = 800. Output = 512.
    W = tf.Variable(tf.truncated_normal([800,512], mean=mu, stddev=sigma))
    B = tf.Variable(tf.zeros(512))
    out = tf.matmul(out, W) + B
    
    # TODO: Activation.
    out = tf.nn.relu(out)

    # TODO: Layer 5: Fully Connected. Input = 512. Output = 512
    W = tf.Variable(tf.truncated_normal([512, 512], mean=mu, stddev=sigma))
    B = tf.Variable(tf.zeros(512))
    out = tf.matmul(out, W) + B
    
    # TODO: Activation.
    out = tf.nn.relu(out)

    # TODO: Layer 5: Fully Connected. Input = 512. Output = 43.
    W = tf.Variable(tf.truncated_normal([512, n_classes], mean=mu, stddev=sigma))
    B = tf.Variable(tf.zeros(n_classes))
    logits = tf.matmul(out, W) + B
    
    return logits

top_k_prediction = tf.nn.top_k(tf.nn.softmax(logits), k = 5)

    sess = tf.get_default_session()
    pred = sess.run(prediction_operation, feed_dict = {x : X_data})

    sess = tf.get_default_session()

def evaluate(X_data, y_data, batch_size):
    num_examples = len(X_data)
    total_accuracy = 0
    sess = tf.get_default_session()
    for offset in range(0, num_examples, batch_size):
        batch_x, batch_y = X_data[offset:offset+batch_size].astype(np.float32), y_data[offset:offset+batch_size]
        accuracy = sess.run(accuracy_operation, feed_dict={x: batch_x, y: batch_y})
        total_accuracy += (accuracy * len(batch_x))
    return total_accuracy / num_examples
